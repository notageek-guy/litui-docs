import { Label } from "@manishxdev/litui";

`Label` is a styled component designed to display labels or tags with customizable styles. It uses `styled-components` and `styled-system` for easy styling and theming.

## Import

```js copy
import { Label } from "@manishxdev/litui";
```

## Example

```jsx
import React from 'react';
import { Label, Input } from "@manishxdev/litui";

const Example = () => {
    return (
        <>
            <Label htmlFor='user'>Username</Label>
            <Input type='text' id='user' />
        </>
    );
}

```
